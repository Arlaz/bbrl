 14where tensors are in shared memory for multiprocessing

 18class SlicedTemporalTensor:
 19    """A SlicedTemporalTensor represents a tensor of size TxBx... by using a list of tensors of size Bx...
 20    The interest is that this tensor automatically adapts its timestep dimension and does not need to have a predefined size.
 21    """
 22
 23    def __init__(self):
 24        """Initialize an empty tensor"""
 25        self.tensors: list[torch.Tensor] = []
 26        self.size: torch.Size = None
 27        self.device: torch.device = None
 28        self.dtype: torch.dtype = None
 29
 30    def set(self, t: int, value: torch.Tensor, batch_dims: Optional[tuple[int, int]]):
 31        """Set a value (dim Bx...) at time t"""
 32        assert (
 33            batch_dims is None
 34        ), "Unable to use batch dimensions with SlicedTemporalTensor"
 35        if self.size is None:
 36            self.size = value.size()
 37            self.device = value.device
 38            self.dtype = value.dtype
 39        assert self.size == value.size(), (
 40            "Incompatible size:" + str(self.size) + " vs " + str(value.size())
 41        )
 42        assert self.device == value.device, "Incompatible device"
 43        assert self.dtype == value.dtype, "Incompatible type"
 44        while len(self.tensors) <= t:
 45            self.tensors.append(
 46                torch.zeros(*self.size, device=self.device, dtype=self.dtype)
 47            )
 48        self.tensors[t] = value
 49
 50    def to(self, device: torch.device):
 51        """Move the tensor to a specific device"""
 52        s = SlicedTemporalTensor()
 53        for k in range(len(self.tensors)):
 54            s.set(k, self.tensors[k].to(device))
 55        return s
 56
 57    def get(self, t: int, batch_dims: Optional[tuple[int, int]]):
 58        """Get the value of the tensor at time t"""
 59
 60        assert (
 61            batch_dims is None
 62        ), "Unable to use batch dimensions with SlicedTemporalTensor"
 63        assert t < len(self.tensors), "Temporal index out of bounds"
 64        return self.tensors[t]
 65
 66    def get_full(self, batch_dims):
 67        """Returns the complete tensor of size TxBx..."""
 68
 69        assert (
 70            batch_dims is None
 71        ), "Unable to use batch dimensions with SlicedTemporalTensor"
 72        return torch.cat([a.unsqueeze(0) for a in self.tensors], dim=0)
 73
 74    def get_time_truncated(
 75        self, from_time: int, to_time: int, batch_dims: Optional[tuple[int, int]]
 76    ):
 77        """Returns tensor[from_time:to_time]"""
 78        assert from_time >= 0 and to_time >= 0 and to_time > from_time
 79        assert batch_dims is None
 80        return torch.cat(
 81            [
 82                self.tensors[k].unsqueeze(0)
 83                for k in range(from_time, min(len(self.tensors), to_time))
 84            ],
 85            dim=0,
 86        )
 87
 88    def set_full(self, value: torch.Tensor, batch_dims: Optional[tuple[int, int]]):
 89        """Set the tensor given a BxTx... tensor.
 90        The input tensor is cut into slices that are stored in a list of tensors
 91        """
 92        assert (
 93            batch_dims is None
 94        ), "Unable to use batch dimensions with SlicedTemporalTensor"
 95        for t in range(value.size()[0]):
 96            self.set(t, value[t], batch_dims=batch_dims)
 97
 98    def time_size(self):
 99        """
100        Return the size of the time dimension
101        """
102        return len(self.tensors)
103
104    def batch_size(self):
105        """Return the size of the batch dimension"""
106        return self.tensors[0].size()[0]
107
108    def select_batch(self, batch_indexes: torch.LongTensor):
109        """Return the tensor where the batch dimension has been selected by the index"""
110        var = SlicedTemporalTensor()
111        for t, v in enumerate(self.tensors):
112            batch_indexes = batch_indexes.to(v.device)
113            var.set(t, v[batch_indexes], None)
114        return var
115
116    def clear(self):
117        """Clear the tensor"""
118        self.tensors = []
119        self.size = None
120        self.device = None
121        self.dtype = None
122
123    def copy_time(self, from_time: int, to_time: int, n_steps: int):
124        """Copy temporal slices of the tensor from from_time:from_time+n_steps to to_time:to_time+n_steps"""
125        for t in range(n_steps):
126            v = self.get(from_time + t, batch_dims=None)
127            self.set(to_time + t, v, batch_dims=None)
128
129    def subtime(self, from_t: int, to_t: int):
130        """
131        Return tensor[from_t:to_t]
132
133        """
134        return CompactTemporalTensor(
135            torch.cat([a.unsqueeze(0) for a in self.tensors[from_t:to_t]], dim=0)
136        )
137
138    def zero_grad(self):
139        """Clear any gradient information in the tensor"""
140        self.tensors = [v.detach() for v in self.tensors]

143class CompactTemporalTensor:
144    """A CompactTemporalTensor is a tensor of size TxBx...
145    It behaves like the `SlicedTemporalTensor` but has a fixed size that cannot change.
146    It is faster than the SlicedTemporalTensor.
147        See `SlicedTemporalTensor`
148    """
149
150    def __init__(self, value: torch.Tensor = None):
151        self.size = None
152        self.device = None
153        self.dtype = None
154        self.tensor = None
155        if value is not None:
156            self.tensor = value
157            self.device = value.device
158            self.size = value.size()
159            self.dtype = value.dtype
160
161    def set(self, t, value, batch_dims):
162        assert False
163        assert self.tensor is not None, "Tensor must be initialized"
164        assert self.size[1:] == value.size(), "Incompatible size"
165        assert self.device == value.device, "Incompatible device"
166        assert self.dtype == value.dtype, "Incompatible type"
167        assert t < self.tensor.size()[0], "Temporal index out of bounds"
168        if batch_dims is None:
169            self.tensor[t] = value
170        else:
171            self.tensor[t, batch_dims[0] : batch_dims[1]] = value
172
173    def select_batch(self, batch_indexes):
174        v = CompactTemporalTensor(self.tensor[:, batch_indexes])
175        return v
176
177    def to_sliced(self) -> SlicedTemporalTensor:
178        """Transform the tensor to a `SlicedTemporalTensor`"""
179        v = SlicedTemporalTensor()
180        for t in range(self.tensor.size()[0]):
181            v.set(t, self.tensor[t], None)
182        return v
183
184    def to(self, device):
185        if device == self.tensor.device:
186            return self
187        t = self.tensor.to(device)
188        return CompactTemporalTensor(t)
189
190    def get(self, t, batch_dims):
191        assert t < self.tensor.size()[0], "Temporal index out of bounds"
192        if batch_dims is None:
193            return self.tensor[t]
194        else:
195            return self.tensor[t, batch_dims[0] : batch_dims[1]]
196
197    def get_full(self, batch_dims):
198        if batch_dims is None:
199            return self.tensor
200        else:
201            return self.tensor[:, batch_dims[0] : batch_dims[1]]
202
203    def time_size(self):
204        return self.tensor.size()[0]
205
206    def batch_size(self):
207        return self.tensor.size()[1]
208
209    def set_full(self, value, batch_dims):
210        if self.tensor is None:
211            assert batch_dims is None
212            self.size = value.size()
213            self.dtype = value.dtype
214            self.device = value.device
215        if batch_dims is None:
216            self.tensor = value
217        else:
218            self.tensor[:, batch_dims[0] : batch_dims[1]] = value
219
220    def subtime(self, from_t, to_t):
221        return CompactTemporalTensor(self.tensor[from_t:to_t])
222
223    def clear(self):
224        self.size = None
225        self.device = None
226        self.dtype = None
227        self.tensor = None
228
229    def copy_time(self, from_time, to_time, n_steps):
230        self.tensor[to_time : to_time + n_steps] = self.tensor[
231            from_time : from_time + n_steps
232        ]
233
234    def zero_grad(self):
235        self.tensor = self.tensor.detach()

238class CompactSharedTensor:
239    """It corresponds to a tensor in shared memory.
240    It is used when building a workspace shared by multiple processes.
241        All the methods behaves like the methods of `SlicedTemporalTensor`
242    """
243
244    def __init__(self, _tensor: torch.Tensor):
245        self.tensor = _tensor
246        self.tensor.share_memory_()
247
248    def set(self, t, value, batch_dims):
249        if batch_dims is None:
250            self.tensor[t] = value.detach()
251        else:
252            self.tensor[t, batch_dims[0] : batch_dims[1]] = value.detach()
253
254    def get(self, t, batch_dims):
255        assert t < self.tensor.size()[0], "Temporal index out of bounds"
256        if batch_dims is None:
257            return self.tensor[t]
258        else:
259            return self.tensor[t, batch_dims[0] : batch_dims[1]]
260
261    def to(self, device):
262        if device == self.tensor.device:
263            return self
264        t = self.tensor.to(device)
265        t.share_memory_()
266        return CompactSharedTensor(t)
267
268    def select_batch(self, batch_indexes):
269        v = CompactSharedTensor(self.tensor[:, batch_indexes])
270        return v
271
272    def get_full(self, batch_dims):
273        if batch_dims is None:
274            return self.tensor
275        else:
276            return self.tensor[:, batch_dims[0] : batch_dims[1]]
277
278    def time_size(self):
279        return self.tensor.size()[0]
280
281    def batch_size(self):
282        return self.tensor.size()[1]
283
284    def set_full(self, value, batch_dims):
285        if batch_dims is None:
286            self.tensor = value.detach()
287        else:
288            self.tensor[:, batch_dims[0] : batch_dims[1]] = value.detach()
289
290    def clear(self):
291        assert False, "Cannot clear a shared tensor"
292
293    def subtime(self, from_t, to_t):
294        t = self.tensor[from_t:to_t]
295        return CompactSharedTensor(t)
296
297    def copy_time(self, from_time, to_time, n_steps):
298        self.tensor[to_time : to_time + n_steps] = self.tensor[
299            from_time : from_time + n_steps
300        ]
301
302    def zero_grad(self):
303        pass

315class Workspace:
316    """Workspace is the most important class in `bbrl`.
317    It correponds to a collection of tensors
318    ('SlicedTemporalTensor`, `CompactTemporalTensor` or ` CompactSharedTensor`).
319    In the majority of cases, we consider that all the tensors have the same time and batch sizes
320    (but it is not mandatory for most of the functions)
321    """
322
323    def __init__(self, workspace: Optional[Workspace] = None):
324        """Create an empty workspace
325
326        Args:
327            workspace (Workspace, optional): If specified, it creates a copy of the workspace
328        (where tensors are cloned as CompactTemporalTensors)
329        """
330        self.variables = {}
331        self.is_shared = False
332        if workspace is not None:
333            for k in workspace.keys():
334                self.set_full(k, workspace[k].clone())
335
336    def set(
337        self,
338        var_name: str,
339        t: int,
340        v: torch.Tensor,
341        batch_dims: Optional[tuple[int, int]] = None,
342    ):
343        """Set the variable var_name at time t"""
344        if var_name not in self.variables:
345            assert not self.is_shared, "Cannot add new variable into a shared workspace"
346            self.variables[var_name] = SlicedTemporalTensor()
347        elif isinstance(self.variables[var_name], CompactTemporalTensor):
348            self.variables[var_name] = self.variables[var_name].to_sliced()
349
350        self.variables[var_name].set(t, v, batch_dims=batch_dims)
351
352    def get(
353        self, var_name: str, t: int, batch_dims: Optional[tuple[int, int]] = None
354    ) -> torch.Tensor:
355        """Get the variable var_name at time t"""
356        assert var_name in self.variables, "Unknown variable '" + var_name + "'"
357        return self.variables[var_name].get(t, batch_dims=batch_dims)
358
359    def clear(self, name=None):
360        """Remove all the variables from the workspace"""
361        if name is None:
362            for k, v in self.variables.items():
363                v.clear()
364            self.variables = {}
365        else:
366            self.variables[name].clear()
367            del self.variables[name]
368
369    def contiguous(self) -> Workspace:
370        """Generates a workspace where all tensors are stored in the Compact format."""
371        workspace = Workspace()
372        for k in self.keys():
373            workspace.set_full(k, self.get_full(k))
374        return workspace
375
376    def set_full(
377        self,
378        var_name: str,
379        value: torch.Tensor,
380        batch_dims: Optional[tuple[int, int]] = None,
381    ):
382        """Set variable var_name with a complete tensor (TxBx...) where T is the time dimension
383        and B is the batch size
384        """
385        if var_name not in self.variables:
386            assert not self.is_shared, "Cannot add new variable into a shared workspace"
387            self.variables[var_name] = CompactTemporalTensor()
388        self.variables[var_name].set_full(value, batch_dims=batch_dims)
389
390    def get_full(
391        self, var_name: str, batch_dims: Optional[tuple[int, int]] = None
392    ) -> torch.Tensor:
393        """Return the complete tensor for var_name"""
394        assert var_name in self.variables, (
395            "[Workspace.get_full] unknown variable '" + var_name + "'"
396        )
397        return self.variables[var_name].get_full(batch_dims=batch_dims)
398
399    def keys(self):
400        """Return an iterator over the variables names"""
401        return self.variables.keys()
402
403    def __getitem__(self, key):
404        """If key is a string, then it returns a torch.Tensor
405        If key is a list of string, it returns a tuple of torch.Tensor
406        """
407        if isinstance(key, str):
408            return self.get_full(key, None)
409        else:
410            return (self.get_full(k, None) for k in key)
411
412    def _all_variables_same_time_size(self) -> bool:
413        """Check that all variables have the same time size"""
414        _ts = None
415        for k, v in self.variables.items():
416            if _ts is None:
417                _ts = v.time_size()
418            if _ts != v.time_size():
419                return False
420        return True
421
422    def time_size(self) -> int:
423        """Return the time size of the variables in the workspace"""
424        _ts = None
425        for k, v in self.variables.items():
426            if _ts is None:
427                _ts = v.time_size()
428            assert _ts == v.time_size(), "Variables must have the same time size"
429        return _ts
430
431    def batch_size(self) -> int:
432        """Return the batch size of the variables in the workspace"""
433        _bs = None
434        for k, v in self.variables.items():
435            if _bs is None:
436                _bs = v.batch_size()
437            assert _bs == v.batch_size(), "Variables must have the same batch size"
438        return _bs
439
440    def select_batch(self, batch_indexes: torch.LongTensor) -> Workspace:
441        """Given a tensor of indexes, returns a new workspace
442        with the selected elements (over the batch dimension)
443        """
444        _bs = None
445        for k, v in self.variables.items():
446            if _bs is None:
447                _bs = v.batch_size()
448            assert _bs == v.batch_size(), "Variables must have the same batch size"
449
450        workspace = Workspace()
451        for k, v in self.variables.items():
452            v = v.select_batch(batch_indexes)
453            workspace.variables[k] = v
454        return workspace
455
456    def select_batch_n(self, n):
457        """Return a new Workspace of batch_size==n by randomly sampling over the batch dimensions"""
458        who = torch.randint(low=0, high=self.batch_size(), size=(n,))
459        return self.select_batch(who)
460
461    def copy_time(
462        self,
463        from_time: int,
464        to_time: int,
465        n_steps: int,
466        var_names: Optional[list[str]] = None,
467    ):
468        """Copy all the variables values from time `from_time` to `from_time+n_steps`
469        to `to_time` to `to_time+n_steps`
470        It can be restricted to specific variables using `var_names`.
471        """
472        for k, v in self.variables.items():
473            if var_names is None or k in var_names:
474                v.copy_time(from_time, to_time, n_steps)
475
476    def get_time_truncated(
477        self,
478        var_name: str,
479        from_time: int,
480        to_time: int,
481        batch_dims: Optional[tuple[int, int]] = None,
482    ) -> torch.Tensor:
483        """Return workspace[var_name][from_time:to_time]"""
484        assert from_time >= 0 and to_time >= 0 and to_time > from_time
485
486        v = self.variables[var_name]
487        if isinstance(v, SlicedTemporalTensor):
488            return v.get_time_truncated(from_time, to_time, batch_dims)
489        else:
490            return v.get_full(batch_dims)[from_time:to_time]
491
492    def get_time_truncated_workspace(self, from_time: int, to_time: int) -> Workspace:
493        """Return a workspace where all variables are truncated between from_time and to_time"""
494        workspace = Workspace()
495        for k in self.keys():
496            workspace.set_full(k, self.get_time_truncated(k, from_time, to_time, None))
497        return workspace
498
499    # Static function
500    def cat_batch(self, workspaces: list[Workspace]) -> Workspace:
501        """Concatenate multiple workspaces over the batch dimension.
502        The workspaces must have the same time dimension.
503        """
504
505        ts = None
506        for w in workspaces:
507            if ts is None:
508                ts = w.time_size()
509            assert ts == w.time_size(), "Workspaces must have the same time size"
510
511        workspace = Workspace()
512        for k in workspaces[0].keys():
513            vals = [w[k] for w in workspaces]
514            v = torch.cat(vals, dim=1)
515            workspace.set_full(k, v)
516        return workspace
517
518    def copy_n_last_steps(self, n: int, var_names: Optional[list[str]] = None) -> None:
519        """Copy the n last timesteps of each variable to the n first timesteps."""
520        _ts = None
521        for k, v in self.variables.items():
522            if var_names is None or k in var_names:
523                if _ts is None:
524                    _ts = v.time_size()
525                assert _ts == v.time_size(), (
526                    "Variables must have the same time size: "
527                    + str(_ts)
528                    + " vs "
529                    + str(v.time_size())
530                )
531
532        for k, v in self.variables.items():
533            if var_names is None or k in var_names:
534                self.copy_time(_ts - n, 0, n)
535
536    def zero_grad(self) -> None:
537        """Remove any gradient information"""
538        for k, v in self.variables.items():
539            v.zero_grad()
540
541    def to(self, device: torch.device) -> Workspace:
542        """Return a workspace where all tensors are on a particular device"""
543        workspace = Workspace()
544        for k, v in self.variables.items():
545            workspace.variables[k] = v.to(device)
546        return workspace
547
548    def _convert_to_shared_workspace(self, n_repeat=1, time_size=None):
549        """INTERNAL METHOD.
550        It converts a workspace to a shared workspace, by repeating this workspace n times on the batch dimension
551        It also automatically adapts the time_size if specified (used in NRemoteAgent.create)
552        """
553
554        with torch.no_grad():
555            workspace = Workspace()
556            for k, v in self.variables.items():
557                value = v.get_full(None).detach()
558                if time_size is not None:
559                    s = value.size()
560                    value = torch.zeros(
561                        time_size, *s[1:], dtype=value.dtype, device=value.device
562                    )
563                ts = [value for _ in range(n_repeat)]
564                value = torch.cat(ts, dim=1)
565                workspace.variables[k] = CompactSharedTensor(value)
566                workspace.is_shared = True
567        return workspace
568
569    def subtime(self, from_t: int, to_t: int) -> Workspace:
570        """
571        Return a workspace restricted to a subset of the time dimension
572        """
573        assert (
574            self._all_variables_same_time_size()
575        ), "All variables must have the same time size"
576        workspace = Workspace()
577        for k, v in self.variables.items():
578            workspace.variables[k] = v.subtime(from_t, to_t)
579        return workspace
580
581    def remove_variable(self, var_name: str):
582        """Remove a variable from the Workspace"""
583        del self.variables[var_name]
584
585    def __str__(self):
586        r = ["Workspace:"]
587        for k, v in self.variables.items():
588            r.append(
589                "\t"
590                + k
591                + ": time_size = "
592                + str(v.time_size())
593                + ", batch_size = "
594                + str(v.batch_size())
595            )
596        return "\n".join(r)

598    def select_subtime(self, t: torch.LongTensor, window_size: int) -> Workspace:
599        """
600        `t` is a tensor of size `batch_size` that provides one time index for each element of the workspace.
601        Then the function returns a new workspace by aggregating `window_size` timesteps starting from index `t`
602        This methods allows to sample multiple windows in the Workspace.
603        Note that the function may be quite slow.
604        """
605        _vars = {k: v.get_full(batch_dims=None) for k, v in self.variables.items()}
606        workspace = Workspace()
607        for k, v in _vars.items():
608            workspace.set_full(
609                k, take_per_row_strided(v, t, num_elem=window_size), batch_dims=None
610            )
611        return workspace

614    def sample_subworkspace(self, n_times, n_batch_elements, n_timesteps):
615        """Sample a workspace from the  workspace. The process is the following:
616                * Let us consider that workspace batch_size is B and time_size is T
617                * For n_times iterations:
618                    * We sample a time window of size n_timesteps
619                    * We then sample a n_batch_elements elements on the batch size
620                    * =>> we obtain a worspace of size n_batch_elements x n_timesteps
621                * We concatenate all the workspaces collected (over the batch dimension)
622
623        Args:
624            n_times ([type]): The number of sub workspaces to sample (and concatenate)
625            n_batch_elements ([type]): <=workspace.batch_size(): nb of batch elements to sample for each subworkspace
626            n_timesteps ([type]): <=workspace.time_size(): the number of timesteps to keep
627
628        Returns:
629            [Workspace]: The resulting workspace
630        """
631        b = self.batch_size()
632        t = self.time_size()
633        to_aggregate = []
634        for _ in range(n_times):
635            assert not n_timesteps > t
636            mini_workspace = self
637            if n_timesteps < t:
638                t = np.random.randint(t - n_timesteps)
639                mini_workspace = self.subtime(t, t + n_timesteps)
640
641            # Batch sampling
642            if n_batch_elements < b:
643                idx_envs = torch.randperm(b)[:n_batch_elements]
644                mini_workspace = mini_workspace.select_batch(idx_envs)
645            to_aggregate.append(mini_workspace)
646
647        if len(to_aggregate) > 1:
648            mini_workspace = Workspace.cat_batch(to_aggregate)
649        else:
650            mini_workspace = to_aggregate[0]
651        return mini_workspace

659            [step1,step2],
660            [step2,step3]
661            ...
662        ]
663        Filters every transitions [step_final,step_initial]
664        """
665        transitions = {}
666        done = self["env/done"][:-1]
667        for key in self.keys():
668            array = self[key]
669
670            # remove transitions (s_terminal -> s_initial)
671            x = array[:-1][~done]
672            x_next = array[1:][~done]

616        """Sample a workspace from the  workspace. The process is the following:
617                * Let us consider that workspace batch_size is B and time_size is T
618                * For n_times iterations:
619                    * We sample a time window of size n_timesteps
620                    * We then sample a n_batch_elements elements on the batch size
621                    * =>> we obtain a worspace of size n_batch_elements x n_timesteps
622                * We concatenate all the workspaces collected (over the batch dimension)
623
624        Args:
625            n_times ([type]): The number of sub workspaces to sample (and concatenate)
626            n_batch_elements ([type]): <=workspace.batch_size(): nb of batch elements to sample for each subworkspace
627            n_timesteps ([type]): <=workspace.time_size(): the number of timesteps to keep
628
629        Returns:
630            [Workspace]: The resulting workspace
631        """
632        b = self.batch_size()
633        t = self.time_size()
634        to_aggregate = []
635        for _ in range(n_times):
636            assert not n_timesteps > t
637            mini_workspace = self
638            if n_timesteps < t:
639                t = np.random.randint(t - n_timesteps)
640                mini_workspace = self.subtime(t, t + n_timesteps)
641
642            # Batch sampling
643            if n_batch_elements < b:
644                idx_envs = torch.randperm(b)[:n_batch_elements]
645                mini_workspace = mini_workspace.select_batch(idx_envs)
646            to_aggregate.append(mini_workspace)
647
648        if len(to_aggregate) > 1:
649            mini_workspace = Workspace.cat_batch(to_aggregate)
650        else:
651            mini_workspace = to_aggregate[0]
652        return mini_workspace
