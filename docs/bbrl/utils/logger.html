244    def to_dataframe(self, with_hps=False):
245        max_len = np.max([len(k) for v, k in self.values.items()])
246        nv = {}
247        for k, v in self.values.items():
248            nnv = [None] * (max_len - len(v))
249            nv[k] = v + nnv
250        self.values = nv
251        it = list(np.arange(max_len))
252        d = {**self.values, **{"iteration": it}}
253        _pd = pd.DataFrame(d)
254        if with_hps:
255            for k in self.hps:
256                _pd["_hp/" + k] = self.hps[k]
257        return _pd

262    def get(self, name, keep_none=False):

269    def replace_none_(self, name):
270        v = self.values[name]
271        last_v = None
272        first_v = None
273        r = []
274        for k in range(len(v)):
275            if v[k] is None:
276                r.append(last_v)
277            else:
278                r.append(v[k])
279                if last_v is None:
280                    first_v = v[k]
281                last_v = v[k]
282
283        p = 0
284        while r[p] is None:
285            r[p] = first_v
286            p += 1
287        self.values[name] = r

289    def max(self, name):

299    def argmin(self, name):
300        v = self.values[name]
301        vv = [k for k in v if k is not None]
302        _max = np.max(vv)
303
304        for k in range(len(v)):
305            if v[k] is None:
306                vv.append(_max + 1.0)
307            else:
308                vv.append(v[k])
309        return np.argmin(vv)
310
311    def argmax(self, name):
312        v = self.values[name]
313        vv = [k for k in v if k is not None]
314        _min = np.min(vv)
315        vv = []
316        for k in range(len(v)):
317            if v[k] is None:
318                vv.append(_min - 1.0)
319            else:
320                vv.append(v[k])
321        return np.argmax(vv)

396def flattify(d):
397    r = {}
398    for k, v in d.items():
399        if isinstance(v, dict):
400            rr = flattify(v)
401            rrr = {k + "/" + kk: rrr for kk, rrr in rr.items()}
402            r = {**r, **rrr}
403        elif isinstance(v, list):
404            r[k] = str(v)
405        else:
406            r[k] = v

410def read_log(directory, use_bz2=True, debug=False):
411    if use_bz2:
412        picklename = directory + "/db.pickle.bzip2"
413        f = bz2.BZ2File(picklename, "rb")
414    else:
415        picklename = directory + "/db.pickle"
416        f = open(picklename, "rb")
417    values = {}
418
419    try:
420        while True:
421            a = pickle.load(f)
422            if a is not None:
423                for name, iteration, value in a:
424                    # print(name,iteration,value)
425                    if debug:
426                        print(name, value, type(value))
427                    if isinstance(value, np.int64):
428                        value = int(value)
429                    if (
430                        isinstance(value, int)
431                        or isinstance(value, float)
432                        or isinstance(value, str)
433                    ):
434                        if name not in values:
435                            values[name] = []
436                        while len(values[name]) < iteration + 1:
437                            values[name].append(None)
438                        values[name][iteration] = value
439    except pickle.PickleError:
440        f.close()
441    f = open(directory + "/params.pickle", "rb")
442    params = pickle.load(f)
443    params = eval(str(params))
444    params = flattify(params)
445    f.close()
446    log = Log(params, values)
447    log.from_directory = directory
448    # f=open(directory+"/fast.pickle","wb")
449    # pickle.dump(log,f)
450    # f.close()
451
452    return log
